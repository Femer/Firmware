#!nsh

# Radio MAVLink start
#

#mavlink start -r 3 -d /dev/ttyS1 -x
mavlink start -d /dev/ttyS1 -x
mavlink stream -d /dev/ttyS1 -s WIND_SAILING_MSG -r 1
usleep 100000
mavlink stream -d /dev/ttyS1 -s GUID_DEBUG_MSG -r 4
usleep 100000
mavlink stream -d /dev/ttyS1 -s PATH_PLAN_MSG -r 1
usleep 100000

# Exit shell to make it available to MAVLink
exit
